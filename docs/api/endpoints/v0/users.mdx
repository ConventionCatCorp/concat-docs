# Users

<splitColumn>
  <div>
    This object represents an end-user within an event organization. Users can be attendees, vendors, performers, staff, etc.
  </div>
  <restList>
    <restElement method='GET' path='/api/v0/users/search' anchor='search-users' />
  </restList>
</splitColumn>

### Search users

<splitColumn>
  <div>
    <p>
      Search for users using a variety of filters. A maximum of 100 results will be returned per request. If additional results are available, they can be retrieved using the <code>nextPage</code> parameter from the previous response.
    </p>
    <h4>Authentication</h4>
    <div>
      <link to="/docs/api/gettingstarted/oauth">Service Integration</link> with <code>user:read</code> scope.
    </div>
    <div style={{ marginBottom: '1.5rem' }} />
    <h4>Request</h4>
    <attribute id="su-nextPage" name="nextPage" type="string" optional>
      A cursor for pagination across multiple pages of results. Don't include this attribute on the first call. Use the <code>nextPage</code> parameter from the previous response for each subsequent request.
    </attribute>
    <attribute id="su-limit" name="limit" type="number" optional>
      The maximum number of results to return, between 1 and 100. Defaults to 100.
    </attribute>
    <parentAttribute id="su-filter" name="filter" type="object" optional>
      <span>Filters to apply to the search.</span>
      <>
        <attribute id="su-flr-roleId" name="roleId" type="number" optional>
          The ID of the role to filter by.
        </attribute>
      </>
    </parentAttribute>
    <div style={{ marginBottom: '1.5rem' }} />
    <h4>Response</h4>
    <parentAttribute id="su-resp-data" name="data" type="Array of User" response optional>
      <span>The users that match the search.</span>
      <>
        <attribute id="su-resp-id" name="id" type="string" response>
          The unique identifier of the user.
        </attribute>
        <attribute id="su-resp-email" name="email" type="string" response>
          The email address of the user.
        </attribute>
        <attribute id="su-resp-preferredName" name="preferredName" type="string" response optional>
          The preferred name of the user. If provided, this should be used instead of the first and ast name, except where legally required.
        </attribute>
        <attribute id="su-resp-firstName" name="firstName" type="string" response>
          The first name of the user.
        </attribute>
        <attribute id="su-resp-lastName" name="lastName" type="string" response>
          The last name of the user.
        </attribute>
        <attribute id="su-resp-username" name="username" type="string" response>
          The username of the user.
        </attribute>
        <attribute id="su-resp-verified" name="verified" type="boolean" response>
          Whether the user has verified their email address.
        </attribute>
        <attribute id="su-resp-phone" name="phone" type="boolean" response>
          The phone number of the user in E.164 format.
        </attribute>
        <parentAttribute id="su-resp-address" name="address" type="Address" response optional>
          <span>The address of the user. If PII is not required to register, the child attributes will be <code>null</code>.</span>
          <>
            <attribute id="su-resp-addr-addressCity" name="addressCity" type="string" response>
              The city of the address.
            </attribute>
            <attribute id="su-resp-addr-addressCountry" name="addressCountry" type="string" response>
              The two-letter country code of the address.
            </attribute>
            <attribute id="su-resp-addr-addressLine1" name="addressLine1" type="string" response>
              The first line of the address.
            </attribute>
            <attribute id="su-resp-addr-addressLine2" name="addressLine2" type="string" response optional>
              The second line of the address, such as an apartment number, if applicable.
            </attribute>
            <attribute id="su-resp-addr-addressState" name="addressState" type="string" response>
              The state or province of the address.
            </attribute>
            <attribute id="su-resp-addr-addressZipcode" name="addressZipcode" type="string" response>
              The ZIP or postal code of the address.
            </attribute>
          </>
        </parentAttribute>
      </>
    </parentAttribute>
    <attribute id="su-resp-nextPage" name="nextPage" type="string" response optional>
      A cursor for pagination across multiple pages of results. If this attribute is present, there are more results available. Use this value in the <code>nextPage</code> parameter for the next request.
    </attribute>
  </div>
