# Dealers

<splitColumn>
  <div>
    This object represents a dealer profile created by a user. Includes business information and submitted applications.
  </div>
  <restList>
    <restElement method='POST' path='/api/v0/dealers/search' anchor='search-dealers' />
  </restList>
</splitColumn>

### Search dealers

<splitColumn>
  <div>
    <p>
      Search for dealers using a variety of filters. A maximum of 100 results will be returned per request. If additional results are available, they can be retrieved using the <code>nextPage</code> parameter from the previous response.
    </p>
    <h4>Authentication</h4>
    <div>
      <link to="/docs/api/gettingstarted/oauth">Service Integration</link> with <code>vendor:read</code> scope.
    </div>
    <div style={{ marginBottom: '1.5rem' }} />
    <h4>Request</h4>
    <attribute id="su-nextPage" name="nextPage" type="string" optional>
      A cursor for pagination across multiple pages of results. Don't include this attribute on the first call. Use the <code>nextPage</code> parameter from the previous response for each subsequent request.
    </attribute>
    <attribute id="su-limit" name="limit" type="number" optional>
      The maximum number of results to return, between 1 and 100. Defaults to 100.
    </attribute>
    <parentAttribute id="su-filter" name="filter" type="object">
      <span>Filters to apply to the search.</span>
      <>
        <attribute id="su-flr-userIds" name="userIds" type="Array of string">
          The IDs of the users to filter by.
        </attribute>
      </>
    </parentAttribute>
    <div style={{ marginBottom: '1.5rem' }} />
    <h4>Response</h4>
    <parentAttribute id="su-resp-data" name="data" type="Array of Dealer" response optional>
      <span>The dealers that match the search.</span>
      <>
        <attribute id="su-resp-id" name="id" type="string" response>
          The unique identifier of the dealer.
        </attribute>
        <attribute id="su-resp-name" name="name" type="string" response>
          The name of the dealer's business.
        </attribute>
        <attribute id="su-resp-email" name="email" type="string" response optional>
          The email address of the dealer's business.
        </attribute>
        <attribute id="su-resp-addressLine1" name="addressLine1" type="string" response>
          The first line of the dealer's business address.
        </attribute>
        <attribute id="su-resp-addressLine2" name="addressLine2" type="string" response optional>
          The second line of the dealer's business address, such as an apartment number, if applicable.
        </attribute>
        <attribute id="su-resp-addressCity" name="addressCity" type="string" response>
          The city of the dealer's business address.
        </attribute>
        <attribute id="su-resp-addressZipcode" name="addressZipcode" type="string" response>
          The ZIP or postal code of the dealer's business address.
        </attribute>
        <attribute id="su-resp-addressState" name="addressState" type="string" response>
          The state or province of the dealer's business address.
        </attribute>
        <attribute id="su-resp-addressCountry" name="addressCountry" type="string" response>
          The two-letter country code of the dealer's business address.
        </attribute>
        <parentAttribute id="su-resp-user" name="user" type="User" response>
          <span>User information for the user who created the dealer entity.</span>
          <>
            <attribute id="su-resp-user-id" name="id" type="string" response>
              The unique identifier of the user.
            </attribute>
            <attribute id="su-resp-user-firstName" name="firstName" type="string" response>
              The first name of the user.
            </attribute>
            <attribute id="su-resp-user-lastName" name="lastName" type="string" response>
              The last name of the user.
            </attribute>
            <attribute id="su-resp-user-username" name="username" type="string" response>
              The username of the user.
            </attribute>
            <attribute id="su-resp-preferredName" name="preferredName" type="string" response optional>
              The preferred name of the user. If provided, this should be used instead of the first and ast name, except where legally required.
            </attribute>
            <attribute id="su-resp-phone" name="phone" type="boolean" response>
              The phone number of the user in E.164 format.
            </attribute>
            <attribute id="su-resp-email" name="email" type="string" response>
              The email address of the user.
            </attribute>
          </>
        </parentAttribute>
        <parentAttribute id="su-resp-applications" name="applications" type="Array of Application" response>
          <span>Applications that the dealer has submitted to the event.</span>
          <>
            <attribute id="su-resp-applications-id" name="id" type="string" response>
              The unique identifier of the application.
            </attribute>
            <attribute id="su-resp-applications-specialRequests" name="specialRequests" type="string" response optional>
              User-specified special requests for the application.
            </attribute>
            <attribute id="su-resp-applications-status" name="status" type="string" response>
              The approval status of the application.
            </attribute>
            <attribute id="su-resp-applications-tableNumber" name="tableNumber" type="string" response optional>
              The assigned table number for the application.
            </attribute>
            <parentAttribute id="su-resp-applications-dealerRequestTypes" name="dealerRequestTypes" type="Array of Product" response>
              <span>Application addons selected by the dealer.</span>
              <>
                <attribute id="su-resp-applications-dealerRequestTypes-id" name="id" type="string" response>
                  The unique identifier of the product.
                </attribute>
                <attribute id="su-resp-applications-dealerRequestTypes-name" name="name" type="string" response>
                  Name given to the product.
                </attribute>
                <attribute id="su-resp-applications-dealerRequestTypes-displayName" name="displayName" type="string" response optional>
                  The user-facing name of the product.
                </attribute>
                <attribute id="su-resp-applications-dealerRequestTypes-paid" name="paid" type="boolean" response>
                  Whether the product has been paid for.
                </attribute>
              </>
            </parentAttribute>
            <parentAttribute id="su-resp-applications-tableType" name="tableType" type="Product" response>
              <span>The table type selected by the dealer in the application.</span>
              <>
                <attribute id="su-resp-applications-tableType-id" name="id" type="string" response>
                  The unique identifier of the product.
                </attribute>
                <attribute id="su-resp-applications-tableType-name" name="name" type="string" response>
                  Name given to the product.
                </attribute>
                <attribute id="su-resp-applications-tableType-displayName" name="displayName" type="string" response optional>
                  The user-facing name of the product.
                </attribute>
                <attribute id="su-resp-applications-tableType-paid" name="paid" type="boolean" response>
                  Whether the product has been paid for.
                </attribute>
              </>
            </parentAttribute>
            <parentAttribute id="su-resp-applications-dealerArea" name="dealerArea" type="DealerArea" response>
              <span>The dealer area the application was submitted for.</span>
              <>
                <attribute id="su-resp-applications-dealerArea-id" name="id" type="string" response>
                  The unique identifier of the dealer area.
                </attribute>
                <attribute id="su-resp-applications-dealerArea-name" name="name" type="string" response>
                  Name given to the dealer area.
                </attribute>
              </>
            </parentAttribute>
            <parentAttribute id="su-resp-applications-options" name="options" type="Array of Option" response>
              <span>A set of options, customized by the event, that contain arbitrary data provided by the dealer.</span>
              <>
                <attribute id="su-resp-applications-options-id" name="id" type="string" response>
                  The unique identifier of the option.
                </attribute>
                <attribute id="su-resp-applications-options-name" name="name" type="string" response>
                  The name of the option as defined by the event.
                </attribute>
                <attribute id="su-resp-applications-options-type" name="type" type="string" response>
                  <div>The type of the option as defined by the event.</div>
                  <div>
                    <strong>Possible values</strong>: <code>select</code>, <code>multi</code>, <code>text</code>, <code>number</code>.
                  </div>
                </attribute>
                <attribute id="su-resp-applications-options-value" name="value" type="number, string, or Array of string" response>
                  The value of the option.
                </attribute>
              </>
            </parentAttribute>
          </>
        </parentAttribute>
        <parentAttribute id="su-resp-options" name="options" type="Array of Option" response>
          <span>A set of options, customized by the event, that contain arbitrary data provided by the dealer.</span>
          <>
            <attribute id="su-resp-options-id" name="id" type="string" response>
              The unique identifier of the option.
            </attribute>
            <attribute id="su-resp-options-name" name="name" type="string" response>
              The name of the option as defined by the event.
            </attribute>
            <attribute id="su-resp-options-type" name="type" type="string" response>
              <div>The type of the option as defined by the event.</div>
              <div>
                <strong>Possible values</strong>: <code>select</code>, <code>multi</code>, <code>text</code>, <code>number</code>, <code>phone</code>, <code>address</code>.
              </div>
            </attribute>
            <attribute id="su-resp-options-value" name="value" type="number, string, or Array of string" response>
              The value of the option.
            </attribute>
          </>
        </parentAttribute>
      </>
    </parentAttribute>
    <attribute id="su-resp-nextPage" name="nextPage" type="string" response optional>
      A cursor for pagination across multiple pages of results. Use this value in the <code>nextPage</code> parameter for the next request.
    </attribute>
    <attribute id="su-resp-hasMore" name="hasMore" type="boolean" response>
      If true, there are more results available. Use the value returned in the <code>nextPage</code> field for the next request.
    </attribute>
  </div>
